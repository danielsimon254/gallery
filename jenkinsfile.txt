pipeline {
    agent any  // Runs on any available agent
    
    environment {
        NODE_ENV = 'production'
        HEROKU_APP_NAME = 'danielsimon-app'  // Replace with your actual Heroku app name
        HEROKU_API_KEY = credentials('HRKU-f5bd0574-c37f-4e52-9fbf-c9cc10470167') // Reference to your stored Heroku API Key
    }

    stages {
        stage('Install Heroku CLI') {
            steps {
                script {
                    // Install Heroku CLI if it's not installed
                    echo "Installing Heroku CLI..."
                    sh '''
                    if ! command -v heroku &> /dev/null; then
                        echo "Heroku CLI not found, installing..."
                        curl https://cli-assets.heroku.com/install.sh | sh
                    else
                        echo "Heroku CLI already installed"
                    fi
                    '''
                }
            }
        }

        stage('Install Dependencies') {
            steps {
                script {
                    // Install Node.js dependencies
                    echo "Installing dependencies..."
                    sh 'npm install'
                }
            }
        }

        stage('Build') {
            steps {
                script {
                    // Build your application (optional for frontend apps like React)
                    echo "Building the application..."
                    sh 'npm run build'
                }
            }
        }

        stage('Deploy to Heroku') {
            steps {
                script {
                    // Authenticate with Heroku using the API key
                    echo "Authenticating with Heroku..."
                    sh '''
                    echo $HEROKU_API_KEY | heroku auth:token
                    git remote remove heroku || true  // Remove any existing Heroku remote if it exists
                    git remote add heroku https://git.heroku.com/$HEROKU_APP_NAME.git
                    git push heroku master  // Push to the Heroku master branch
                    '''
                }
            }
        }
    }

    post {
        always {
            echo "Pipeline execution finished."
        }
    }
}
